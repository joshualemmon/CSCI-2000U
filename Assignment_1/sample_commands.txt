echo hello world
	-outputs hello world in the terminal

echo {con,pre}{sent,fer}{s,ed}
	-outputs all the combinations of the arguments, eg con+sent+s or pre+fer+s
	-full output is consents consented confers confered presents presented prefers prefered

date
	-outputs the current date and time

whoami
	-outputs the current user's name

cal 2000
	-outputs the calender for the year 2000

cal 9 1752
	-outputs the calender for september in the year 1752
	-missing days 3-13 due to the change in calender type

bc -l
	-outputs copyright and warranty information for the os

echo 5+4 | bc -l
	-outputs 9
